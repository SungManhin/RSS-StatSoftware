is.even<-function(x)
{
  return(x%%2==0)
}

## [1] TRUE
  • 给出输入数字的最大值和最小值
MinMax<-function(x)
{
  m1 = min(x)
  m2 = max(x)
  return(c(m1, m2))
}
MinMax(rnorm(100))
## [1] -2.415497  2.363047
  • 给出输入数字的分位数值、均值和标准差
statistic<-function(x)
{
  quant = quantile(x)
  Mean = mean(x)
  SD = sd(x)
  return(list(quant = quant, Mean = Mean, SD = SD))
}
